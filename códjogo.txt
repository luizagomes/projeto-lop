var x = 100
var y = 200

function setup(){
createCanvas(400, 400);
}

function draw(){
background(255);
//elipse representa o jogador
ellipse(x, y, 30, 30);
if(keyIsDown(LEFT_ARROW)){
x-=5
}
if(keyIsDown(RIGHT_ARROW)){
x+=5;
}
if(keyIsDown(UP_ARROW)){
y-=30;
}
if(keyIsDown(DOWN_ARROW)){
y+=30;
}

//quadrado representa o obstaculo
square(200, 165, 50);
}
